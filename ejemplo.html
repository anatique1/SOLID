Interface Segregation Principle

codigo mal hecho 


<!-- ERROR: Etiqueta div no cierra, comillas faltantes y falta de semántica -->
<div class=contenedor-error
    <div>Título de la página (No es H1)</div>
    <p>Este texto no tiene etiqueta de cierre.

<!-- VIOLACIÓN DE SOLID (Interface Segregation) -->
interface IAve {
    void volar();
    void nadar();
    void comer();
}

// MAL: El Avestruz se ve obligada a implementar volar() aunque no vuela
class Avestruz implements IAve {
    public void volar() { /* Error: No puedo volar */ }
    public void nadar() { /* ... */ }
    public void comer() { /* ... */ }
}

codigo bien hecho 


<!-- CORRECCIÓN: Etiquetas cerradas, comillas y semántica correcta -->
<header class="contenedor-ok">
    <h1>Título de la página</h1>
    <p>Este texto sí está bien estructurado.</p>
</header>

<!-- CUMPLIMIENTO DE SOLID (Interface Segregation) -->
interface IVolador { void volar(); }
interface INadador { void nadar(); }
interface IComedor { void comer(); }

// BIEN: El Avestruz solo implementa lo que realmente hace
class Avestruz implements INadador, IComedor {
    public void nadar() { /* ... */ }
    public void comer() { /* ... */ }
}
